  match b with

Definition kername_of_qualid (q : qualid) : TemplateMonad kername :=
  gr <- tmLocate1 q ;;
  match gr with
  | ConstRef kn ⇒ ret kn
  | IndRef ind ⇒ ret ind.(inductive_mind)
  | ConstructRef ind _ ⇒ ret ind.(inductive_mind)
  | VarRef _ ⇒ tmFail ("tmLocate: " ^ q ^ " is a Var")
