Deriving a compact induction principle for terms

Allows to get the right induction principle on lists of terms appearing in the term syntax (in evar, applications, branches of cases and (co-)fixpoints.


Custom induction principle on syntax, dealing with the various lists appearing in terms.

Lemma term_forall_list_ind :
  ∀ P : term → Type,
    (∀ n : nat, P (tRel n)) →
    (∀ i : ident, P (tVar i)) →
    (∀ (n : nat) (l : list term), All P l → P (tEvar n l)) →
    (∀ s, P (tSort s)) →
    (∀ (n : aname) (t : term), P t → ∀ t0 : term, P t0 → P (tProd n t t0)) →
    (∀ (n : aname) (t : term), P t → ∀ t0 : term, P t0 → P (tLambda n t t0)) →
    (∀ (n : aname) (t : term),
        P t → ∀ t0 : term, P t0 → ∀ t1 : term, P t1 → P (tLetIn n t t0 t1)) →
    (∀ t u : term, P t → P u → P (tApp t u)) →
    (∀ s (u : list Level.t), P (tConst s u)) →
    (∀ (i : inductive) (u : list Level.t), P (tInd i u)) →
    (∀ (i : inductive) (n : nat) (u : list Level.t), P (tConstruct i n u)) →
    (∀ (ind : case_info) (p : predicate term),
        tCasePredProp P P p → ∀ c : term, P c → ∀ l : list (branch term),
        tCaseBrsProp P l → P (tCase ind p c l)) →
    (∀ (s : projection) (t : term), P t → P (tProj s t)) →
    (∀ (m : mfixpoint term) (n : nat), tFixProp P P m → P (tFix m n)) →
    (∀ (m : mfixpoint term) (n : nat), tFixProp P P m → P (tCoFix m n)) →
    
    ∀ t : term, P t.

Lemma term_forall_ctx_list_ind :
  ∀ (P : context → term → Type),
    (∀ Γ (n : nat), P Γ (tRel n)) →
    (∀ Γ (i : ident), P Γ (tVar i)) →
    (∀ Γ (n : nat) (l : list term), All (P Γ) l → P Γ (tEvar n l)) →
    (∀ Γ s, P Γ (tSort s)) →
    (∀ Γ (n : aname) (t : term), P Γ t → ∀ t0 : term, P (vass n t :: Γ) t0 → P Γ (tProd n t t0)) →
    (∀ Γ (n : aname) (t : term), P Γ t → ∀ t0 : term, P (vass n t :: Γ) t0 → P Γ (tLambda n t t0)) →
    (∀ Γ (n : aname) (t : term),
        P Γ t → ∀ t0 : term, P Γ t0 → ∀ t1 : term, P (vdef n t t0 :: Γ) t1 → P Γ (tLetIn n t t0 t1)) →
    (∀ Γ (t u : term), P Γ t → P Γ u → P Γ (tApp t u)) →
    (∀ Γ s (u : list Level.t), P Γ (tConst s u)) →
    (∀ Γ (i : inductive) (u : list Level.t), P Γ (tInd i u)) →
    (∀ Γ (i : inductive) (n : nat) (u : list Level.t), P Γ (tConstruct i n u)) →
    (∀ Γ (ci : case_info) (p : predicate term) (t : term) (brs : list (branch term)),
        CasePredProp P Γ p →
        P Γ t →
        CaseBrsProp p P Γ brs →
        P Γ (tCase ci p t brs)) →
    (∀ Γ (s : projection) (t : term), P Γ t → P Γ (tProj s t)) →
    (∀ Γ (m : mfixpoint term) (n : nat),
        All_local_env (on_local_decl (fun Γ' t ⇒ P (Γ ,,, Γ') t)) (fix_context m) →
        tFixProp (P Γ) (P (Γ ,,, fix_context m)) m → P Γ (tFix m n)) →
    (∀ Γ (m : mfixpoint term) (n : nat),
        All_local_env (on_local_decl (fun Γ' t ⇒ P (Γ ,,, Γ') t)) (fix_context m) →
        tFixProp (P Γ) (P (Γ ,,, fix_context m)) m → P Γ (tCoFix m n)) →
    
    ∀ Γ (t : term), P Γ t.

Lemma term_ind_size_app :
  ∀ (P : term → Type),
    (∀ (n : nat), P (tRel n)) →
    (∀ (i : ident), P (tVar i)) →
    (∀ (n : nat) (l : list term), All (P) l → P (tEvar n l)) →
    (∀ s, P (tSort s)) →
    (∀ (n : aname) (t : term), P t → ∀ t0 : term, P t0 → P (tProd n t t0)) →
    (∀ (n : aname) (t : term), P t → ∀ t0 : term, P t0 → P (tLambda n t t0)) →
    (∀ (n : aname) (t : term),
        P t → ∀ t0 : term, P t0 → ∀ t1 : term, P t1 →
                                                   P (tLetIn n t t0 t1)) →
    (∀ (t u : term),
        (∀ t', size t' < size (tApp t u) → P t') →
        P t → P u → P (tApp t u)) →
    (∀ s (u : list Level.t), P (tConst s u)) →
    (∀ (i : inductive) (u : list Level.t), P (tInd i u)) →
    (∀ (i : inductive) (n : nat) (u : list Level.t), P (tConstruct i n u)) →
    (∀ (ci : case_info) (p : PCUICAst.predicate term) (c : term) (brs : list (branch term)),
        tCasePredProp P P p → P c →
        tCaseBrsProp P brs → P (tCase ci p c brs)) →
    (∀ (s : projection) (t : term), P t → P (tProj s t)) →
    (∀ (m : mfixpoint term) (n : nat),
        tFixProp P P m → P (tFix m n)) →
    (∀ (m : mfixpoint term) (n : nat),
        tFixProp (P) P m → P (tCoFix m n)) →
    
    ∀ (t : term), P t.
