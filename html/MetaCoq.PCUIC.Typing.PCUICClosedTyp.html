Lemma term_closedn_list_ind :
  ∀ (P : nat → term → Type),
    (∀ k (n : nat), n < k → P k (tRel n)) →
    (∀ k (i : ident), P k (tVar i)) →
    (∀ k (n : nat) (l : list term), All (P k) l → P k (tEvar n l)) →
    (∀ k s, P k (tSort s)) →
    (∀ k (n : aname) (t : term), P k t → ∀ t0 : term, P (S k) t0 → P k (tProd n t t0)) →
    (∀ k (n : aname) (t : term), P k t → ∀ t0 : term, P (S k) t0 → P k (tLambda n t t0)) →
    (∀ k (n : aname) (t : term),
        P k t → ∀ t0 : term, P k t0 → ∀ t1 : term, P (S k) t1 → P k (tLetIn n t t0 t1)) →
    (∀ k (t u : term), P k t → P k u → P k (tApp t u)) →
    (∀ k s (u : list Level.t), P k (tConst s u)) →
    (∀ k (i : inductive) (u : list Level.t), P k (tInd i u)) →
    (∀ k (i : inductive) (n : nat) (u : list Level.t), P k (tConstruct i n u)) →
    (∀ k (ci : case_info) (p : predicate term),
        tCasePredProp_k P k p →
        ∀ t0 : term, P k t0 → ∀ l : list (branch term),
        tCaseBrsProp_k P p k l → P k (tCase ci p t0 l)) →
    (∀ k (s : projection) (t : term), P k t → P k (tProj s t)) →
    (∀ k (m : mfixpoint term) (n : nat), tFixProp (P k) (P (#|fix_context m| + k)) m → P k (tFix m n)) →
    (∀ k (m : mfixpoint term) (n : nat), tFixProp (P k) (P (#|fix_context m| + k)) m → P k (tCoFix m n)) →
    
    ∀ k (t : term), closedn k t → P k t.

Lemma term_noccur_between_list_ind :
  ∀ (P : nat → nat → term → Type),
    (∀ k n (i : nat), i < k ∨ k + n ≤ i → P k n (tRel i)) →
    (∀ k n (i : ident), P k n (tVar i)) →
    (∀ k n (id : nat) (l : list term), All (P k n) l → P k n (tEvar id l)) →
    (∀ k n s, P k n (tSort s)) →
    (∀ k n (na : aname) (t : term), P k n t → ∀ t0 : term, P (S k) n t0 → P k n (tProd na t t0)) →
    (∀ k n (na : aname) (t : term), P k n t → ∀ t0 : term, P (S k) n t0 → P k n (tLambda na t t0)) →
    (∀ k n (na : aname) (t : term),
        P k n t → ∀ t0 : term, P k n t0 → ∀ t1 : term, P (S k) n t1 → P k n (tLetIn na t t0 t1)) →
    (∀ k n (t u : term), P k n t → P k n u → P k n (tApp t u)) →
    (∀ k n s (u : list Level.t), P k n (tConst s u)) →
    (∀ k n (i : inductive) (u : list Level.t), P k n (tInd i u)) →
    (∀ k n (i : inductive) (c : nat) (u : list Level.t), P k n (tConstruct i c u)) →
    (∀ k n (ci : case_info) (p : predicate term),
        tCasePredProp_k (fun k' ⇒ P k' n) k p → ∀ t0 : term, P k n t0 →
        ∀ l : list (branch term),
        tCaseBrsProp_k (fun k' ⇒ P k' n) p k l → P k n (tCase ci p t0 l)) →
    (∀ k n (s : projection) (t : term), P k n t → P k n (tProj s t)) →
    (∀ k n (m : mfixpoint term) (i : nat), tFixProp (P k n) (P (#|fix_context m| + k) n) m → P k n (tFix m i)) →
    (∀ k n (m : mfixpoint term) (i : nat), tFixProp (P k n) (P (#|fix_context m| + k) n) m → P k n (tCoFix m i)) →
    
    ∀ k n (t : term), noccur_between k n t → P k n t.
