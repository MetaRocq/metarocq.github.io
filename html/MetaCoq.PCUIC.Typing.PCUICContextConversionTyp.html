Lemma split_closed_context {Γ : context} (n : nat) :
  is_closed_context Γ →
  n ≤ #|Γ| →
  ∑ (Δ : closed_context) (Δ' : open_context Δ),
    [× Δ = skipn n Γ :> context, Δ' = firstn n Γ :> context,
       Γ = Δ ,,, Δ' & n = #|Δ'|].
